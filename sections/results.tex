	\frametitle{RESULTS}
	\begin{itemize}
		\item From the results, \alert{26.01\%} were stunted and \alert{10.87\%} were underweight. 
		\item A majority of stunted children \alert{(15.08\%)} and underweight children \alert{(7.2\%)} were from \textcolor {blue}{low income households} (lowest and second lowest categories).
		\item \textcolor {blue}{Rural areas had a higher proportion} of malnourished children than urban areas where \alert{19.2\%} of children were Stunted and \alert{8.48\%} were underweight.
		\item \textcolor {blue} {Males had a higher proportion} of stunting \alert{(15.06\%)} and underweight \alert{(6.08\%)} as compared to females who had \alert{10.95\%} stunted and \alert{4.79\%} underweight.
		\end {itemize}
	\end{frame}

	\begin{frame}
		\frametitle {RESULTS cont'd}
		\begin{itemize}
			\item Mothers who attained \textcolor {blue}{primary level of education} were \alert{1.5 times more likely} to have stunted children and \alert{1.7 times more  likely} to have underweight children as compared to those with a \textcolor {blue}{higher level of education}. 
			\item Children born and living in \textcolor {blue}{rural areas} are \alert{1.88 times more likely} to be stunted and \alert{1.83 times more likely} to be underweight as compared to children born and living in \textcolor {blue}{urban areas}. 
			\item Children from the \textcolor {blue}{poorest households} were \alert{2.2 times more likely} to be stunted and \alert{3.23 times more likely} to be underweight than children from the \textcolor {blue} {richest households}. 
			\item Children who were \textcolor {blue}{very small at birth} were \alert{1.6 times more likely} to be stunted than those who were \textcolor {blue}{larger in size  at birth}.
			\item \textcolor {blue}{Maternal age} was also a significant predictor of malnutrition. \textcolor {blue} {Mother’s aged 15-19} were \alert{2.16 times more likely} to give birth to stunted  and underweight children  as compared to \textcolor {blue}{mothers aged 45-49}. 
		\end{itemize}
