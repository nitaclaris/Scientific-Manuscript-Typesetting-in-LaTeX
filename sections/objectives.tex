	\begin{frame}
		\frametitle{OBJECTIVES}
		\begin {block} {General Objective}
		To determine the factors associated with under-five malnutrition in Kenya using the KDHS 2014 dataset.
		\end {block}
		\begin {block} {Specific Objectives}
		\begin{itemize}
			\item To estimate the prevalence of malnutrition among children under-five in Kenya.
			\item To determine the association between bio-demographic and socio-economic factors and under-five malnutrition.
			\item To fit a multivariate model to determine factors associated with under-five malnutrition.
		\end{itemize}
		\end{block}
	\end{frame}
