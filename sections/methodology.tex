	\begin{frame}
		\frametitle{METHODOLOGY}
		\begin{itemize}
			\item The study used the Kenya Demographic Health Survey (KDHS 2014) dataset. Specifically, the interest was with the Children’s recode \alert{(KEKR72FL)}.
			\item Descriptive statistics were done to summarize the data quantitatively.
			\item The summary statistics also gave a general outlook of the data. 
			\item \textcolor {blue}{Bivariate analysis} was done to examine the strengths of the relationship among variables of interest.
			\item For \textcolor {blue} {multivariate analysis}, the model applied was the \alert{Modified Poisson Regression model}. 
		\end{itemize}
	\end{frame}
